        window.copyObjectivePreventive = function(id) {
            const item = assignments.find(a => a.id === id);
            if (!item) return;

            const formatDateTH = (d) => {
                if (isNaN(d.getTime())) return '-';
                const day = String(d.getDate()).padStart(2, '0');
                const month = String(d.getMonth() + 1).padStart(2, '0');
                const year = d.getFullYear() + 543;
                return `${day}/${month}/${year}`;
            };

            let dAction = new Date();
            if (item.actionDate) dAction = new Date(item.actionDate);
            const dateActionStr = formatDateTH(dAction);
            
            let dInform = new Date(dAction);
            dInform.setDate(dInform.getDate() - 1);
            const dateInformStr = formatDateTH(dInform);
            
            let dReceive = new Date(dAction);
            dReceive.setDate(dReceive.getDate() - 2);
            const dateReceiveStr = formatDateTH(dReceive);

            const text = `1.ลักษณะงาน : ${item.jobType || '-'}\n2.หน่วยงานที่แจ้ง : ${item.agency || '-'}\n3.สถานที่ : ${item.location || '-'}\n4.จำนวนเสา : ${item.objective || '-'}\n5.วันที่รับแจ้ง : ${dateReceiveStr}\n6.วันที่แจ้ง ผรม. : ${dateInformStr}\n7.วันที่เข้าดำเนินการ : ${dateActionStr}\n8.ระยะเวลาดำเนินการ : 1 วัน \n9.Code Route : ${item.routeCode}\n10.GPS : ${item.gps}`;
            
            // เปลี่ยนเป็นแสดง Popup เพื่อให้ผู้ใช้ดูข้อมูลก่อนแล้วค่อยกด Copy
            Swal.fire({
                title: 'สรุป Objective',
                html: `<textarea class="w-full border border-gray-300 rounded p-3 text-sm font-prompt bg-gray-50 focus:outline-none" rows="9" readonly>${text}</textarea>`,
                showCancelButton: true,
                confirmButtonText: '<i class="fa-regular fa-copy mr-1"></i> Copy',
                confirmButtonColor: '#3b82f6',
                cancelButtonText: 'ปิด',
                customClass: { container: 'font-sans', popup: 'rounded-2xl' }
            }).then((result) => {
                if (result.isConfirmed) {
                    fallbackCopyTextToClipboard(text);
                }
            });

        window.signInAzure = async function() {
            if (isLoginInProgress) return;
            try {
                isLoginInProgress = true;
                $('#login-error').addClass('hidden').text('');
                const loginRequest = {
                    scopes: ["User.Read"],
                    prompt: "select_account"
                };
                
                const inIframe = window !== window.parent && !window.opener;
                if (inIframe) {
                    const response = await msalInstance.loginPopup(loginRequest);
                    if (response && response.account) {
                        processLoginSuccess(response.account);
                    }
                } else {
                    msalInstance.loginRedirect(loginRequest);
                }
            } catch (error) {
                console.error("Login Error:", error);
                if (error.errorCode === "user_cancelled") {
                    // Ignore user cancellation
                } else if (error.errorCode === "interaction_in_progress") {
                    // Clear stuck interaction status
                    const keysToRemove = [];
                    for (let i = 0; i < sessionStorage.length; i++) {
                        const key = sessionStorage.key(i);
                        if (key && key.indexOf("interaction.status") !== -1) {
                            keysToRemove.push(key);
                        }
                    }
                    keysToRemove.forEach(k => sessionStorage.removeItem(k));
                    $('#login-error').removeClass('hidden').text('Error: Please try clicking Sign in again.');
                } else {
                    $('#login-error').removeClass('hidden').text('Error: ' + error.message);
                }
                isLoginInProgress = false;
            }

        window.hasPermission = function(requiredKey, type = 'action') {
            if (window.userRoles.includes('Admin')) return true; // Admin gets everything
            
            // Check customized roles
            if (masterData && masterData.rolePermissions) {
                for (let r of window.userRoles) {
                    const roleCfg = masterData.rolePermissions[r];
                    if (roleCfg) {
                        if (type === 'menu' && roleCfg.menus && roleCfg.menus.includes(requiredKey)) return true;
                        if (type === 'action' && roleCfg.actions && roleCfg.actions.includes(requiredKey)) return true;
                    }
                }
            }
            return false;
        };

            // Apply Menus Visibility

                if (menu.id.endsWith('_menu')) {
                    const elId = menu.id.replace('_menu', '-menu');
                    const btn = $(`#${elId}`).prev('button'); 
                    if (hasMenuPerm) {
                        btn.show();
                        // Optional: don't auto show the div, let the toggle handle it
                    } else {
                        btn.hide();
                        $(`#${elId}`).hide(); // also hide the submenu container
                    }
                } else {
                    const el = $(`[data-view="${menu.id}"]`);
                    // Specifically hide parent if it's sub_dashboard (and others) but not affecting siblings if they don't have to.
                    if (hasMenuPerm) el.show(); else el.hide();

        async function initApp() { 
            try { 
                const loadAssignments = async () => {
                    const { data, error } = await supabaseClient.from('assignments').select('*').order('internal_id', { ascending: false });
                    if (error) { console.error("Load Assignments Error:", error); return; }
                    assignments = data.map(r => ({ id: r.id, ...(r.data || {}), status: r.status, createdAt: r.created_at, updatedAt: r.updated_at, internalId: r.internal_id }));
                    window.updateSidebarBadges();
                    window.updateView(); 
                };

                const loadSettings = async () => {
                    const { data, error } = await supabaseClient.from('master_settings').select('*').eq('id', 'masterData').single();
                    if (error && error.code !== 'PGRST116') { console.error("Load Settings Error:", error); return; }
                    if (data && data.data) {
                        masterData = data.data;
                        if (!masterData.rolePermissions) masterData.rolePermissions = {}; // Make sure it exists
                        window.renderDropdowns(); 
                        window.applyRolePermissions();
                        if (currentView === 'setting') window.renderSettings(); 
                    } else {
                        if (typeof initializeDatabaseMaster === 'function') initializeDatabaseMaster(false);
                    }
                };

                const loadLinkSupport = async () => {
                    const { data, error } = await supabaseClient.from('link_support').select('*').order('created_at', { ascending: true });
                    if (error) { console.error("Load Link Support Error:", error); return; }
                    linkSupportList = data || [];
                    if (currentView === 'link_support') window.renderLinkSupport();
                };

                // Initial Load
                loadAssignments();
                loadSettings();
                loadLinkSupport();

                // Supabase Real-time Subscriptions
                supabaseClient.channel('assignments_channel')
                    .on('postgres_changes', { event: '*', schema: 'public', table: 'assignments' }, payload => {
                        loadAssignments();
                    }).subscribe();

                supabaseClient.channel('settings_channel')
                    .on('postgres_changes', { event: '*', schema: 'public', table: 'master_settings' }, payload => {
                        loadSettings();
                    }).subscribe();

                supabaseClient.channel('link_support_channel')
                    .on('postgres_changes', { event: '*', schema: 'public', table: 'link_support' }, payload => {
                        loadLinkSupport();
                    }).subscribe();

            } catch (e) { 
                console.error("Init Error:", e); 
                Swal.fire('Connection Error', e.message, 'error'); 
            } 
        }

            if(sumDashSelect.length) {
                CONST_DATA.jobTypes.summary_plan.forEach(t => {
                    sumDashSelect.append(new Option(t, t));
                });
            }

            // Logic เปลี่ยนกล่อง Round Robin ตาม Job Type
            $('#form-job-type').change(function() {
                const val = $(this).val();
                const isSubMenu = menuConfig[currentCategoryFilter]?.isSub;
                if (isSubMenu) {
                    if (val === 'Special Job') {
                        $('#rr-normal-view').addClass('hidden');
                        $('#rr-special-view').removeClass('hidden');
                        // Reset sub name so user must select manually
                        $('#form-sub-name').val($('#form-manual-sub').val()); 
                    } else {
                        $('#rr-normal-view').removeClass('hidden');
                        $('#rr-special-view').addClass('hidden');
                        
                        // FIX: ตรวจสอบว่าอยู่ในโหมดแก้ไขหรือไม่
                        const docId = $('#doc-id').val();
                        if (!docId) {
                             // ถ้าเป็นงานใหม่ ให้ดึงคิวล่าสุด
                             fetchRoundRobinState();
                        } else {
                             // ถ้าแก้ไข ให้คืนค่าเดิมที่เคยบันทึกไว้ (ไม่ดึงคิวใหม่)
                             const originalSub = $('#form-sub-name').data('original');
                             if(originalSub) {
                                 $('#form-sub-name').val(originalSub);
                                 $('#rr-next-name').text(originalSub);
                             }
                        }
                    }
                }
            });

            // Sym Impact Other Toggle
            $('#form-sym-impact').change(function() {
                if($(this).val() === 'Other') $('#form-sym-impact-other').addClass('show');
                else $('#form-sym-impact-other').removeClass('show');
            });

            // Link Support Type Toggle
            $('#ls-type').change(function() {
                if($(this).val() === 'Other') {
                    $('#ls-type-other').removeClass('hidden').addClass('block').focus();
                } else {
                    $('#ls-type-other').addClass('hidden').removeClass('block');
                }
            });
        });
        
    </script>
